/*
|--------------------------------------------------------------------------
| Knowledge Base Controller Routes
|--------------------------------------------------------------------------
|
| Endpoint: /knowledgebase
|

Route::group(['prefix' => '/knowledgebase'], function () {

    Route::get('/', 'KnowledgebaseController@index')->name('admin.knowledgebase');
    Route::post('/settings', 'KnowledgebaseController@settings')->name('admin.knowledgebase.settings');

    Route::group(['prefix' => '/questions'], function () {
        Route::get('/', 'KnowledgebaseController@question_index')->name('admin.knowledgebase.questions.index');
        Route::get('/new', 'KnowledgebaseController@question_new')->name('admin.knowledgebase.questions.new');
        Route::post('/create', 'KnowledgebaseController@question_create')->name('admin.knowledgebase.questions.create');
        Route::get('/edit/{id}', 'KnowledgebaseController@question_edit')->name('admin.knowledgebase.questions.edit');
        Route::post('/update/{id}', 'KnowledgebaseController@question_update')->name('admin.knowledgebase.questions.update');
        Route::delete('/delete', 'KnowledgebaseController@question_delete')->name('admin.knowledgebase.questions.delete');
    });

    Route::group(['prefix' => '/category'], function () {
        Route::get('/', 'KnowledgebaseController@category_index')->name('admin.knowledgebase.category.index');
        Route::get('/new', 'KnowledgebaseController@category_new')->name('admin.knowledgebase.category.new');
        Route::post('/create', 'KnowledgebaseController@category_create')->name('admin.knowledgebase.category.create');
        Route::get('/edit/{id}', 'KnowledgebaseController@category_edit')->name('admin.knowledgebase.category.edit');
        Route::post('/update/{id}', 'KnowledgebaseController@category_update')->name('admin.knowledgebase.category.update');
        Route::delete('/delete', 'KnowledgebaseController@category_delete')->name('admin.knowledgebase.category.delete');
    });
});
